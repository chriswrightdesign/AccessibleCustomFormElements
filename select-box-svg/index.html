The CSS - Component

                
/* Component */

/*
Optional reset of auto-zoom
What this snippet does is: stop the automatic focus zoom on small screen devices,
this doesn't take away the option to zoom,
instead it leaves it up to the user.
*/
@media all and (max-width:60em) {
    select:focus,
    select:hover {
        font-size:1em;
    }
}
.select-box-wrapper-svg input::-ms-clear,
.select-box-wrapper-svg input::-ms-reveal {
    display:none; /* Aggressive Reset for IE */
}

.select-box-label {
    display:block;
    margin:1em 0;
}

/* Select box wrapper */
.select-box-wrapper-svg {
    border: none;
    border-radius: 5px;
    font-family: Helvetica, sans-serif;
    font-size: 1em;
    line-height: 1em;
    margin: 0;
    overflow: hidden;
    position: relative;
    transition: background-color 0.4s ease-in-out;
    width: 100%;
}
.select-box-wrapper-svg:hover,
.select-box-wrapper-svg:focus,
.select-box-wrapper-svg:active {
    cursor: pointer;
}

/* Position the SVG arrow into the wrapper */
.svg-arrow {
    z-index:1;
    position:absolute;
    right:0;
    bottom:0;
    display:block;
    padding:0;

}

.select-box-wrapper-svg select option {
    display:block;
    line-height:1.5;
    font-size:1em;
    border:none;
}
.select-box-wrapper-svg select:hover {
    cursor: pointer;
}
.select-box-wrapper-svg select {
    position:relative;
    z-index:2;
    -webkit-appearance: none;
    -moz-appearance: none;
    appearance: none;
    background-color: transparent;
    border-color:none;
    background-image: none;
    border: none;
    box-shadow: none;
    font-family: Helvetica, sans-serif;
    font-size: 1em;
    padding: 16px;
    display: block;
    transition: background 0.4s ease;

}

/* This is a simple way of prequalifying for IE8 and below,
if the browser doesn't support MQs, just use the standard select arrow */
@media all {
    .select-box-wrapper-svg select {
        width: 115%;
    }
}
